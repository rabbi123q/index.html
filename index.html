import requests # Step 1: Create a new email address response = requests.post('https://api.mail.tm/accounts', json={'address': 'example@mail.tm', 'password': 'password123'}) email_info = response.json() email_address = email_info['address'] # Step 2: Retrieve the inbox for the created email inbox_response = requests.get(f'https://api.mail.tm/messages/{email_address}') inbox_info = inbox_response.json() # Display inbox info print(inbox_info)
